//--------------------------------------------------
// 元件(component)的範例
//--------------------------------------------------
interface component
{
    // 回傳元件的標籤(string)，[must]
    get tag() {}

    // 在上層綁定操作介面，提供給其他元件使用，[optional]
    _bind(root) {}

    // 提供載入、儲存的功能，[optional]
    load(data) {}
    save() {}
}
